```kotlin
//В Kotlin по умолчанию все классы считаются final, т.е от них нельзя наследоваться
//Чтобы можно было наследоваться от класса нужно приписать ключевое слово open
open class Animal(val name: String, var weight: Float, val habitat: String);
//И вызвать конструктор родительского класса 
class Dog(weight: Float) : Animal("Dog", weight, "Area"){}
//в данном случае weight является параметром конструктора класса Dog, а не полноценным свойством этого же класса, т.к свойство с таким же именем уже есть в классе Animal

//Также чтобы переопределять методы родительского класса необходимо их помечать ключевым словом open
open fun eat(){  
    println("Eating food")  
}

//И если мы хотим переопределить метод из род. класса, то мы в дочернем классе используем ключевое слово override
override fun eat(){  
    println("Eating kost")  
}
```
