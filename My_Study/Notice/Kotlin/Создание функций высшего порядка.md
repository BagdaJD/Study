fun modifyString(str: String, modify: (String) -> String) : String{  
    return modify(str)  
}  
//Это функция высшего порядка, т.к принимает в качестве аргумента другую функцию
//modify - это передаваемая функция, которая принимает строку и возвращает строку

inline fun modifyList(list: List<Int>, modify: (List<Int>) -> Int) : Int{  
    return modify(list)  
}  
val res = modifyList(list){it.sum()}
//Здесь аналогично
//сами функции сы передаем в основной программе, в качестве анонимных

//inline - нужен для того, чтобы при вызове наших таких функций
//не создавался анонимный объект с таким методом
//чтобы не тратить столько ресурсов, для оптимизации пишем inline
fun func(i: Int, mod: (Int) -> Unit){  
    println(mod(i))  
}  
//Если наша функция ничего не возвращает, и мы передаем в качестве параеметра другую функцию, которая тоже ничего не возвращает, то Unit в качестве возвращаемого значения необязательно писать

fun main(){  
    val list: List<Int> = listOf(11, 12, 13, 14, 15)  
    val res = modifyList(list){it.sum()}  
    println(res)  
}
