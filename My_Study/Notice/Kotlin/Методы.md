//Cat.kt
class Cat(val name: String, val age: Int, val weight: Float) {  
    fun printInfo(){  
        println("Name = $name; Age = $age; Weight = $weight;")  
    }  
    fun checkAgeCat() : Boolean = age >= 12  
}

//main.kt
fun main(){  
    val cat = Cat("Barsik", 12, 4f)  
    cat.printInfo()  
    print(cat.checkAgeCat())  
}
//Примеры методов того или иного класса 


fun Cat.isOld() = age >= 12;

//Полезный пример 
//Cat.kt
val isOld: Boolean = age >= 12;

cat.printInfo()  
println(cat.isOld)  
cat.age = 10  
print(cat.isOld)

//Чтобы программа работала корректно нужно сделать следующее

val isOld: Boolean
