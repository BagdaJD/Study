Дано целое число _N_ (> 0). Найти сумму
1/1 + 1/2 + 1/3 .... 1/n

val N = readLine().toInt  
var sum: Double = 0  
  
for i <- 1 to N do  
  sum += 1.0/i  
  println(s"i = $i; sum = $sum")  
println(sum)
//Если мы знаем ,что в результате программы будут возникать дробные числа, то переменная ,должна быть вещественной и операции должны проходить ,хотя бы с одним вещественным числом
